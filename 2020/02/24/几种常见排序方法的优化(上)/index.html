    /**

    public static <K extends Comparable<K>> void sort(K[] a) {
        int h = 1, len = a.length;

        // æ„å»ºæ’æ’çš„æ­¥é•¿h

        while (h < len / step) h = step * h + 1;

                }
            }
            h /= step;
        }
    }
