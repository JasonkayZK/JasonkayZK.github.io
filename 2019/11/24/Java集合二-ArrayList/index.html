    boolean batchRemove(Collection<?> c, boolean complement,
                        final int from, final int end) {
        Objects.requireNonNull(c);
        final Object[] es = elementData;
        int r;
        // Ë∑≥ËøáÊï∞ÁªÑÂâç‰∏çÈúÄË¶ÅÊìç‰ΩúÁöÑÂÖÉÁ¥†(Optimize for initial run of survivors)

            if (r == end)
                return false;
            if (c.contains(es[r]) != complement)
                break;
        }
        // Ê∑ªÂä†Êï∞ÁªÑÊåáÈíà
        int w = r++;
        // ‰∏ãÈù¢ÁöÑÁÆóÊ≥ïÊÄùË∑ØÊòØ: ‰ΩøÁî®wËÆ∞ÂΩï‰∏ã‰∏Ä‰∏™‰øùÂ≠òÁöÑ‰ΩçÁΩÆ, Â¶ÇÊûúÈâ¥ÂÆöÂÖÉÁ¥†Êª°Ë∂≥complementÊù°‰ª∂, ÂàôÊèíÂÖ•w‰ΩçÁΩÆ, ÊúÄÂêéÈÄöËøáshiftTailOverGap()ÊñπÊ≥ï,Â∞ÜÊú´Â∞æÂÖÉÁ¥†Ê∏ÖÁ©∫
        try {
            for (Object e; r < end; r++)
                if (c.contains(e = es[r]) == complement)
                    es[w++] = e;
        } catch (Throwable ex) {
            // Âç≥‰Ωøc.contains()ÊäõÂá∫ÂºÇÂ∏∏Ôºå‰πüË¶Å‰øùÊåÅ‰∏éAbstractCollectionÁöÑË°å‰∏∫ÂÖºÂÆπÊÄß(Preserve behavioral compatibility with AbstractCollection, even if c.contains() throws)
            System.arraycopy(es, r, es, w, end - r);
            w += end - r;
            throw ex;
        } finally {
            modCount += end - w;
            shiftTailOverGap(es, w, end);
        }
        return true;
    }
