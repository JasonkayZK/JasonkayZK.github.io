     * always collide. (Among known examples are sets of Float keys
     * holding consecutive whole numbers in small tables.)

     * downward. There is a tradeoff between speed, utility, and
     * quality of bit-spreading. Because many common sets of hashes
     * are already reasonably distributed (so don't benefit from
     * spreading), and because we use trees to handle large sets of
     * collisions in bins, we just XOR some shifted bits in the

    final V putVal(int hash, K key, V value, boolean onlyIfAbsent,

    /**
    * ÂàùÂßãÂåñÊàñËÄÖÊâ©ÂÆπ‰πãÂêéÂÖÉÁ¥†Ë∞ÉÊï¥
    * Â¶ÇÊûú‰∏∫Á©∫ÔºåÂàôÊ†πÊçÆÂ≠óÊÆµÈòàÂÄº‰∏≠‰øùÁïôÁöÑÂàùÂßãÂÆπÈáèÁõÆÊ†áËøõË°åÂàÜÈÖç(16)

    *
     * Initializes or doubles table size.  If null, allocates in
     * accord with initial capacity target held in field threshold.
     * Otherwise, because we are using power-of-two expansion, the
     * elements from each bin must either stay at same index, or move
     * with a power of two offset in the new table.
     *
     * @return the table
     */
    final Node<K,V>[] resize() {
        // Ëé∑ÂèñÊóßÂÖÉÁ¥†Êï∞ÁªÑÁöÑÂêÑÁßç‰ø°ÊÅØ
        Node<K,V>[] oldTab = table;
        // ÊóßË°®ÂÆπÈáè
        int oldCap = (oldTab == null) ? 0 : oldTab.length;
        // ÊóßË°®Êâ©ÂÆπÁöÑ‰∏¥ÁïåÂÄº
        int oldThr = threshold;
        // ÂÆö‰πâÊñ∞Êï∞ÁªÑÁöÑÈïøÂ∫¶ÂèäÊâ©ÂÆπÁöÑ‰∏¥ÁïåÂÄº
        int newCap, newThr = 0;
        if (oldCap > 0) {  // Â¶ÇÊûúÂéütable‰∏ç‰∏∫Á©∫
            if (oldCap >= MAXIMUM_CAPACITY) { 
                // Â¶ÇÊûúÊï∞ÁªÑÈïøÂ∫¶ËææÂà∞ÊúÄÂ§ßÂÄºÔºåÂàô‰øÆÊîπ‰∏¥ÁïåÂÄº‰∏∫Integer.MAX_VALUE
                threshold = Integer.MAX_VALUE;   
                return oldTab;
            }
            // ‰∏ãÈù¢Â∞±ÊòØÊâ©ÂÆπÊìç‰ΩúÔºà2ÂÄçÔºâ
            else if ((newCap = oldCap << 1) < MAXIMUM_CAPACITY &&
                     oldCap >= DEFAULT_INITIAL_CAPACITY)
                newThr = oldThr << 1; // threshold‰πüÂèò‰∏∫‰∫åÂÄç
        }
        else if (oldThr > 0) // initial capacity was placed in threshold
            newCap = oldThr;
        else {               // threshold‰∏∫0ÔºåÂàô‰ΩøÁî®ÈªòËÆ§ÂÄº
            newCap = DEFAULT_INITIAL_CAPACITY;
            newThr = (int)(DEFAULT_LOAD_FACTOR * DEFAULT_INITIAL_CAPACITY);
        }
        if (newThr == 0) { // Â¶ÇÊûú‰∏¥ÁïåÂÄº‰πü‰∏∫0ÔºåÂàôËÆæÁΩÆ‰∏¥ÁïåÂÄº
            float ft = (float)newCap * loadFactor;
            newThr = (newCap < MAXIMUM_CAPACITY && ft < (float)MAXIMUM_CAPACITY ?
                      (int)ft : Integer.MAX_VALUE);
        }
        threshold = newThr; // Êõ¥Êñ∞Â°´ÂÖÖÂõ†Â≠ê
        @SuppressWarnings({"rawtypes","unchecked"})
        Node<K,V>[] newTab = (Node<K,V>[])new Node[newCap];
        table = newTab;
        if (oldTab != null) { // Ë∞ÉÊï¥Êï∞ÁªÑÂ§ßÂ∞è‰πãÂêéÔºåÈúÄË¶ÅË∞ÉÊï¥Á∫¢ÈªëÊ†ëÊàñËÄÖÈìæË°®ÁöÑÊåáÂêë
            for (int j = 0; j < oldCap; ++j) {
                Node<K,V> e;
                if ((e = oldTab[j]) != null) {
                    oldTab[j] = null;
                    if (e.next == null)
                        newTab[e.hash & (newCap - 1)] = e;
                    else if (e instanceof TreeNode) // Á∫¢ÈªëÊ†ëË∞ÉÊï¥
                        ((TreeNode<K,V>)e).split(this, newTab, j, oldCap);
                    else { // preserve order
                        // ÈìæË°®Ë∞ÉÊï¥
                        Node<K,V> loHead = null, loTail = null;
                        Node<K,V> hiHead = null, hiTail = null;
                        Node<K,V> next;
                        do {
                            next = e.next;
                            if ((e.hash & oldCap) == 0) {
                                if (loTail == null)
                                    loHead = e;
                                else
                                    loTail.next = e;
                                loTail = e;
                            }
                            else {
                                if (hiTail == null)
                                    hiHead = e;
                                else
                                    hiTail.next = e;
                                hiTail = e;
                            }
                        } while ((e = next) != null);
                        if (loTail != null) {
                            loTail.next = null;
                            newTab[j] = loHead;
                        }
                        if (hiTail != null) {
                            hiTail.next = null;
                            newTab[j + oldCap] = hiHead;
                        }
                    }
                }
            }
        }
        return newTab;
    }

        /*
     * Â¶ÇÊûúÂÖÉÁ¥†Êï∞ÁªÑ‰∏∫Á©∫ ÊàñËÄÖ Êï∞ÁªÑÈïøÂ∫¶Â∞è‰∫é Ê†ëÁªìÊûÑÂåñÁöÑÊúÄÂ∞èÈôêÂà∂

        if (tab == null || (n = tab.length) < MIN_TREEIFY_CAPACITY)
            resize(); // Êâ©ÂÆπ

        // Â¶ÇÊûúÂÖÉÁ¥†Êï∞ÁªÑÈïøÂ∫¶Â∑≤ÁªèÂ§ß‰∫éÁ≠â‰∫é‰∫Ü MIN_TREEIFY_CAPACITYÔºåÈÇ£‰πàÂ∞±ÊúâÂøÖË¶ÅËøõË°åÁªìÊûÑËΩ¨Êç¢‰∫Ü
        // Ê†πÊçÆhashÂÄºÂíåÊï∞ÁªÑÈïøÂ∫¶ËøõË°åÂèñÊ®°ËøêÁÆóÂêéÔºåÂæóÂà∞ÈìæË°®ÁöÑÈ¶ñËäÇÁÇπ
        else if ((e = tab[index = (n - 1) & hash]) != null) {
            TreeNode<K,V> hd = null, tl = null; // ÂÆö‰πâÁ∫¢ÈªëÊ†ëÁöÑÈ¶ñ„ÄÅÂ∞æËäÇÁÇπ
            do {
                // Â∞ÜËØ•ËäÇÁÇπËΩ¨Êç¢‰∏∫ Ê†ëËäÇÁÇπ
                TreeNode<K,V> p = replacementTreeNode(e, null);
                if (tl == null) // Â¶ÇÊûúÂ∞æËäÇÁÇπ‰∏∫Á©∫ÔºåËØ¥ÊòéËøòÊ≤°ÊúâÊ†πËäÇÁÇπ
                    hd = p; // È¶ñËäÇÁÇπÔºàÊ†πËäÇÁÇπÔºâÊåáÂêë ÂΩìÂâçËäÇÁÇπ
                else  // Â∞æËäÇÁÇπ‰∏ç‰∏∫Á©∫Ôºå‰ª•‰∏ã‰∏§Ë°åÊòØ‰∏Ä‰∏™ÂèåÂêëÈìæË°®ÁªìÊûÑ

                }

            } while ((e = e.next) != null);

            // Âà∞ÁõÆÂâç‰∏∫Ê≠¢ ‰πüÂè™ÊòØÊääNodeÂØπË±°ËΩ¨Êç¢Êàê‰∫ÜTreeNodeÂØπË±°ÔºåÊääÂçïÂêëÈìæË°®ËΩ¨Êç¢Êàê‰∫ÜÂèåÂêëÈìæË°®

        int binCount = 0;

    private void writeObject(java.io.ObjectOutputStream s)
        throws IOException {
        int buckets = capacity();
        // Write out the threshold, loadfactor, and any hidden stuff
        s.defaultWriteObject();

        // Write out number of buckets
        s.writeInt(buckets);

        // Write out size (number of Mappings)
        s.writeInt(size);

        internalWriteEntries(s);
    }

    // Â∞ÜHashMapÁöÑ‚ÄúÊÄªÁöÑÂÆπÈáèÔºåÂÆûÈôÖÂÆπÈáèÔºåÊâÄÊúâÁöÑEntry‚Äù‰æùÊ¨°ËØªÂá∫
    private void readObject(java.io.ObjectInputStream s)
        throws IOException, ClassNotFoundException {
        // Read in the threshold (ignored), loadfactor, and any hidden stuff
        s.defaultReadObject();

        // ÂàùÂßãÂåñÂØπË±°
        reinitialize();
        if (loadFactor <= 0 || Float.isNaN(loadFactor))
            throw new InvalidObjectException("Illegal load factor: " +
                                             loadFactor);

        // Read and ignore number of buckets
        s.readInt();           
        // Read number of mappings (size)
        int mappings = s.readInt(); 
        if (mappings < 0)
            throw new InvalidObjectException("Illegal mappings count: " +
                                             mappings);
        else if (mappings > 0) { // (if zero, use defaults)

            // ËÆæÁΩÆloadFactor
            // Size the table using given load factor only if within
            // range of 0.25...4.0
            float lf = Math.min(Math.max(0.25f, loadFactor), 4.0f);
            float fc = (float)mappings / lf + 1.0f;

            // ËÆæÁΩÆÂÆπÈáè
            int cap = ((fc < DEFAULT_INITIAL_CAPACITY) ?
                       DEFAULT_INITIAL_CAPACITY :
                       (fc >= MAXIMUM_CAPACITY) ?
                       MAXIMUM_CAPACITY :
                       tableSizeFor((int)fc));

            // ËÆæÁΩÆÂÆπÈáèÈòàÂÄº
            float ft = (float)cap * lf;
            threshold = ((cap < MAXIMUM_CAPACITY && ft < MAXIMUM_CAPACITY) ?
                         (int)ft : Integer.MAX_VALUE);

            // Ê£ÄÊü•ÂØπË±°ÊòØÂê¶ÊòØMap.Entry[]Á±ªÂûã
            // Check Map.Entry[].class since it's the nearest public type to what we're actually creating
            SharedSecrets.getJavaObjectInputStreamAccess().checkArray(s, Map.Entry[].class, cap);
            @SuppressWarnings({"rawtypes","unchecked"})
            Node<K,V>[] tab = (Node<K,V>[])new Node[cap];
            table = tab;

            // ËØªÂÖ•keyÂíåvalueÂπ∂ÊîæÂÖ•map‰∏≠
            // Read the keys and values, and put the mappings in the HashMap
            for (int i = 0; i < mappings; i++) {
                @SuppressWarnings("unchecked")
                    K key = (K) s.readObject();
                @SuppressWarnings("unchecked")
                    V value = (V) s.readObject();
                putVal(hash(key), key, value, false, false);
            }
        }
    }

    abstract class HashIterator {
        Node<K,V> next;        // next entry to return
        Node<K,V> current;     // current entry
        int expectedModCount;  // for fast-fail
        int index;             // current slot

        // ÊäΩË±°Á±ªÁöÑÊûÑÈÄ†Âô®, ‰æõÂ≠êÁ±ªÁªßÊâø
        HashIterator() {
            expectedModCount = modCount;
            Node<K,V>[] t = table;
            current = next = null;
            index = 0;
            // Âú®ÊûÑÈÄ†Âô®‰∏≠‰ºöÊâæÂà∞Êï∞ÁªÑ‰∏≠Á¨¨‰∏Ä‰∏™‰∏çÊòØÊåáÂêënullÁöÑÈÇ£‰∏™ÂÖÉÁ¥†
            // Â∞ÜËøô‰∏™ÂÖÉÁ¥†Â≠òÂÖ•next‰∏≠ÔºåÂêåÊó∂indexÊåáÂêëËøô‰∏™Êï∞ÁªÑÂÖÉÁ¥†ÁöÑ‰∏ã‰∏Ä‰∏™‰∏ãÊ†á
            if (t != null && size > 0) { // advance to first entry
                do {} while (index < t.length && (next = t[index++]) == null);
            }
        }

        public final boolean hasNext() {
            return next != null;
        }

        final Node<K,V> nextNode() {
            Node<K,V>[] t;
            Node<K,V> e = next;
            if (modCount != expectedModCount)
                throw new ConcurrentModificationException();
            if (e == null)
                throw new NoSuchElementException();
