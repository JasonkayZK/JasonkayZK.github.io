    public Vector(int initialCapacity, int capacityIncrement) {
        super();
        if (initialCapacity < 0)

        this.elementData = new Object[initialCapacity];
        this.capacityIncrement = capacityIncrement;
    }

    // ÊåáÂÆöÈõÜÂêàÁöÑVectorÊûÑÈÄ†ÂáΩÊï∞
    public Vector(Collection<? extends E> c) {
        // Ëé∑ÂèñÈõÜÂêàcÁöÑÊï∞ÁªÑÔºåÂπ∂Â∞ÜÂÖ∂ËµãÂÄºÁªôelementData
        elementData = c.toArray();
        // ËÆæÁΩÆÊï∞ÁªÑÈïøÂ∫¶
        elementCount = elementData.length;
        // defend against c.toArray (incorrectly) not returning Object[]
        if (elementData.getClass() != Object[].class)
            elementData = Arrays.copyOf(elementData, elementCount, Object[].class);

    // Â∞ÜÊï∞ÁªÑVectorÁöÑÂÖ®ÈÉ®ÂÖÉÁ¥†ÈÉΩÊã∑Ë¥ùÂà∞Êï∞ÁªÑanArray‰∏≠
    public synchronized void copyInto(Object[] anArray) {

    public synchronized void trimToSize() {
        modCount++;
        int oldCapacity = elementData.length;
        if (elementCount < oldCapacity) {
            elementData = Arrays.copyOf(elementData, elementCount);
        }
    }

    // Ê£ÄÊü•Âπ∂Á°Æ‰øùVectorÁöÑÂÆπÈáèË∂≥Â§ü
    public synchronized void ensureCapacity(int minCapacity) {
        if (minCapacity > 0) {
            modCount++;
            // ÂΩìVectorÁöÑÂÆπÈáè‰∏çË∂≥‰ª•ÂÆπÁ∫≥ÂΩìÂâçÁöÑÂÖ®ÈÉ®ÂÖÉÁ¥†ÔºåÂ¢ûÂä†ÂÆπÈáèÂ§ßÂ∞è
            if (minCapacity > elementData.length)
                grow(minCapacity);
        }
    }

    // ÂèØÂàÜÈÖçÁöÑÊúÄÂ§ßÁ©∫Èó¥(The maximum size of array to allocate)
    private static final int MAX_ARRAY_SIZE = Integer.MAX_VALUE - 8;

    // Ê†πÊçÆminCapacityÁªôÂ≠òÂÇ®Êï∞ÁªÑÊâ©ÂÆπ
    private Object[] grow(int minCapacity) {
        return elementData = Arrays.copyOf(elementData,
                                           newCapacity(minCapacity));
    }

    // ÁªôÂ≠òÂÇ®Êï∞ÁªÑÊâ©ÂÆπ‰∏Ä‰∏™Âçï‰Ωç
    private Object[] grow() {
        return grow(elementCount + 1);
    }

    // Ê†πÊçÆminCapacityËÆ°ÁÆóÊâ©ÂÆπÂêéÊï∞ÁªÑÂ§ßÂ∞è
    private int newCapacity(int minCapacity) {
        // ÂÆπÊòìËÆ°ÁÆóÊ∫¢Âá∫ÁöÑ‰ª£Á†ÅÂå∫(overflow-conscious code)
        int oldCapacity = elementData.length;

        // Ëã•ÂÆπÈáèÂ¢ûÈáèÁ≥ªÊï∞>0(Âç≥capacityIncrement>0)ÔºåÂàôÂ∞ÜÂÆπÈáèÂ¢ûÂ§ßÂΩìcapacityIncrement
        // Âê¶ÂàôÔºåÂ∞ÜÂÆπÈáèÂ¢ûÂ§ß‰∏ÄÂÄç
        int newCapacity = oldCapacity + ((capacityIncrement > 0) ?
                                         capacityIncrement : oldCapacity);
        if (newCapacity - minCapacity <= 0) {
            if (minCapacity < 0) // ÂèëÁîü‰∫ÜËÆ°ÁÆóÊ∫¢Âá∫(overflow)
                throw new OutOfMemoryError();
            return minCapacity;
        }
        // Êñ∞ÁöÑÂÆπÈáèÂ∞è‰∫éÁ≠â‰∫é Integer.MAX_VALUE - 8 ÂàôÁõ¥Êé•ËøîÂõûÔºåÂê¶ÂàôÂü∫‰∫éÈ¢ÑÊúüÂÆπÈáèËøîÂõû Integer.MAX_VALUE - 8 Êàñ Integer.MAX_VALUE
        return (newCapacity - MAX_ARRAY_SIZE <= 0)
            ? newCapacity
            : hugeCapacity(minCapacity);
    }

     // Âü∫‰∫éÈ¢ÑÊúüÂÆπÈáèËøîÂõû Integer.MAX_VALUE - 8 Êàñ Integer.MAX_VALUE
    private static int hugeCapacity(int minCapacity) {
        if (minCapacity < 0) // overflow
            throw new OutOfMemoryError();
        return (minCapacity > MAX_ARRAY_SIZE) ?
            Integer.MAX_VALUE :
            MAX_ARRAY_SIZE;
    }

    public synchronized void setSize(int newSize) {
        modCount++;
        // Ëã• "newSize Â§ß‰∫é VectorÂÆπÈáè"ÔºåÂàôË∞ÉÊï¥VectorÁöÑÂ§ßÂ∞è
        if (newSize > elementData.length)
            grow(newSize);
        final Object[] es = elementData;

        for (int to = elementCount, i = newSize; i < to; i++)
            es[i] = null;
        elementCount = newSize;
    }

    // Ëã•ÊâæÂà∞ÔºåÂàôËøîÂõûÂÖÉÁ¥†ÁöÑÁ¥¢ÂºïÂÄºÔºõÂê¶ÂàôÔºåËøîÂõû-1

    // Ëã•ÊâæÂà∞ÔºåÂàôËøîÂõûÂÖÉÁ¥†ÁöÑÁ¥¢ÂºïÂÄºÔºõÂê¶ÂàôÔºåËøîÂõû-1
    public synchronized int indexOf(Object o, int index) {
        // Ëã•Êü•ÊâæÂÖÉÁ¥†‰∏∫nullÔºåÂàôÊ≠£ÂêëÊâæÂá∫nullÂÖÉÁ¥†ÔºåÂπ∂ËøîÂõûÂÆÉÂØπÂ∫îÁöÑÂ∫èÂè∑
        if (o == null) {
            for (int i = index ; i < elementCount ; i++)
                if (elementData[i]==null)
                    return i;
        } else {
            // Ëã•Êü•ÊâæÂÖÉÁ¥†‰∏ç‰∏∫nullÔºåÂàôÊ≠£ÂêëÊâæÂá∫ËØ•ÂÖÉÁ¥†ÔºåÂπ∂ËøîÂõûÂÆÉÂØπÂ∫îÁöÑÂ∫èÂè∑
            for (int i = index ; i < elementCount ; i++)
                if (o.equals(elementData[i]))
                    return i;
        }
        return -1;
    }

    // Ëã•ÊâæÂà∞ÔºåÂàôËøîÂõûÂÖÉÁ¥†ÁöÑ‚ÄúÁ¥¢ÂºïÂÄº‚ÄùÔºõÂê¶ÂàôÔºåËøîÂõû-1
    public synchronized int lastIndexOf(Object o, int index) {
        if (index >= elementCount)
            throw new IndexOutOfBoundsException(index + " >= "+ elementCount);

        if (o == null) {
            // Ëã•Êü•ÊâæÂÖÉÁ¥†‰∏∫nullÔºåÂàôÂèçÂêëÊâæÂá∫nullÂÖÉÁ¥†ÔºåÂπ∂ËøîÂõûÂÆÉÂØπÂ∫îÁöÑÂ∫èÂè∑
            for (int i = index; i >= 0; i--)
                if (elementData[i]==null)
                    return i;
        } else {
            // Ëã•Êü•ÊâæÂÖÉÁ¥†‰∏ç‰∏∫nullÔºåÂàôÂèçÂêëÊâæÂá∫ËØ•ÂÖÉÁ¥†ÔºåÂπ∂ËøîÂõûÂÆÉÂØπÂ∫îÁöÑÂ∫èÂè∑
            for (int i = index; i >= 0; i--)
                if (o.equals(elementData[i]))
                    return i;
        }
        return -1;
    }

    public synchronized void removeElementAt(int index) {
        if (index >= elementCount) {
            throw new ArrayIndexOutOfBoundsException(index + " >= " +
                                                     elementCount);
        }
        else if (index < 0) {
            throw new ArrayIndexOutOfBoundsException(index);
        }
        int j = elementCount - index - 1;
        if (j > 0) {
            System.arraycopy(elementData, index + 1, elementData, index, j);
        }
        modCount++;
        elementCount--;
        elementData[elementCount] = null; // GC‰ºòÂåñ(to let gc do its work)
    }

    // Âú®index‰ΩçÁΩÆÂ§ÑÊèíÂÖ•ÂÖÉÁ¥†(obj)
    public synchronized void insertElementAt(E obj, int index) {
        if (index > elementCount) {
            throw new ArrayIndexOutOfBoundsException(index
                                                     + " > " + elementCount);
        }
        modCount++;
        final int s = elementCount;
        Object[] elementData = this.elementData;
        if (s == elementData.length)
            elementData = grow();
        System.arraycopy(elementData, index,
                         elementData, index + 1,
                         s - index);
        elementData[index] = obj;
        elementCount = s + 1;
    }

    // This helper method split out from add(E) to keep method

        elementData[s] = e;
        elementCount = s + 1;

                             numMoved);
        elementData[--elementCount] = null; // ‰ºòÂåñGC(Let gc do its work)

        return oldValue;
    }

    public boolean addAll(Collection<? extends E> c) {
        Object[] a = c.toArray();
        modCount++;
        int numNew = a.length;
        if (numNew == 0)
            return false;
        // VectorÂØπÊ≠§ÊñπÊ≥ïÂÅö‰∫Ü‰ºòÂåñ, JDK 8‰πãÂâçÁöÑÊñπÊ≥ïÊï¥‰∏™Ë¢´Ê†áÊ≥®‰∏∫synchronized
        synchronized (this) {
            Object[] elementData = this.elementData;
            final int s = elementCount;
            if (numNew > elementData.length - s)
                elementData = grow(s + numNew);
            // Â∞ÜÈõÜÂêàcÁöÑÂÖ®ÈÉ®ÂÖÉÁ¥†Êã∑Ë¥ùÂà∞Êï∞ÁªÑelementData‰∏≠

            elementCount = s + numNew;
            return true;
        }
    }

    // JDK 8‰∏≠Ê∑ªÂä†ÁöÑÊñπÊ≥ï, Ê†πÊçÆPredicate(LambdaË°®ËææÂºè)ÁªôÂá∫ÁöÑbooleanÊù°‰ª∂, Âà†Èô§Êª°Ë∂≥Êù°‰ª∂ÁöÑÂÖÉÁ¥†
    // Â§öÁî®Âú®StreamÊìç‰Ωú‰∏≠
    @Override
    public boolean removeIf(Predicate<? super E> filter) {

    /* ‰∏Ä‰∫õ‰ΩçÊìç‰ΩúÂÆûÁé∞(A tiny bit set implementation) */

    // Áî®Âú®‰∫ÜremoveIf()ÊñπÊ≥ï‰∏≠, ‰ΩúÁî®ÊòØÂàùÂßãÂåñ‰∏Ä‰∏™Êï∞ÁªÑ, Êï∞ÁªÑÈïøÂ∫¶ÊòØÊåâÁÖß (n - 1) >> 6 + 1 ÁöÑÈÄªËæëÊù•ÂæóÂà∞ÁöÑ
    // >> ÊòØÁÆóÊúØ‰ΩçÁßªÔºåÁÆÄÂçïÁöÑÊù•ËØ¥Â∞±ÊòØÂ∞ÜÊï∞Â≠óÈô§‰ª• 2 ÁöÑ n Ê¨°ÊñπÔºå>> 6 Â∞±ÊòØÈô§‰ª• 2 ÁöÑ 6 Ê¨°ÊñπÔºåÂç≥Èô§‰ª• 64, Âõ†Ê≠§ nBits ÊñπÊ≥ï‰ºö‰ª• 64 ‰∏∫Âçï‰ΩçÊù•Êî∂Áº©ÔºåÂπ∂ËøîÂõûÊî∂Áº©ÂêéÈïøÂ∫¶ÁöÑÊï∞ÁªÑ
    private static long[] nBits(int n) {
        return new long[((n - 1) >> 6) + 1];
    }

     // ÂêåÊ†∑Áî®Âú®removeIf()ÊñπÊ≥ï‰∏≠, ‰ΩúÁî®Â∞±ÊòØÂ∞Ü i ËΩ¨Êç¢‰∏∫ 2 ÁöÑ i Ê¨°ÊñπÊï∞Â≠óÂêéÔºåËøõË°åÂêàÂπ∂ÔºåËÄåÂêàÂπ∂ÁöÑÁªìÊûúÂÖ∂ÂÆûÊòØÂ∞ÜÁ¨¨ i ‰ΩçÁΩÆ‰∏∫ 1, ÂÖ∂‰Ωô‰Ωç‰∏∫Èõ∂(‰ΩçÂõæÁ¥¢Âºï)
    // Âú®removeIf()ÊñπÊ≥ï‰∏≠: deathRow ÊòØ‰∏Ä‰∏™„Äå‰ΩçÂõæ„Äç(Bitmap) ÁöÑÊï∞ÊçÆÁªìÊûÑ
    private static void setBit(long[] bits, int i) {
        bits[i >> 6] |= 1L << i;
    }

    // ÂêåÊ†∑Áî®Âú®removeIf()ÊñπÊ≥ï‰∏≠: isClear ‰ºöÊåâÁÖßËæìÂÖ•ÂéªÁúãÊï∞ÁªÑ‰∏≠ÂÖÉÁ¥†ÁöÑÁ¨¨ i ‰ΩçÊòØÂê¶Ë¢´ÁΩÆ‰∏∫ 1ÔºåÂ¶ÇÊûú Ê≤°Êúâ Ë¢´ÁΩÆ‰∏∫ 1 ÔºåÂàôËøîÂõû true(Âà§Êñ≠‰ΩçÂõæÁ¥¢Âºï)
    private static boolean isClear(long[] bits, int i) {
        return (bits[i >> 6] & (1L << i)) == 0;
    }

    // JDK 8ÂêéÊñ∞Âä†ÂÖ•, Áî®‰∫é‰º†ÂÖ•ÁªôÂÆöÁöÑPredicate(Ê≠£ÂàôË°®ËææÂºè), Êù•Âà†Èô§Êª°Ë∂≥Êù°‰ª∂ÁöÑÂÖÉÁ¥†
    private synchronized boolean bulkRemove(Predicate<? super E> filter) {
        int expectedModCount = modCount;
        final Object[] es = elementData;
        final int end = elementCount;
        int i;
        // ‰ªéÂ§¥ÂºÄÂßãÁï•Ëøá‰∏çÊª°Ë∂≥Êñ≠Ë®ÄÁöÑÂÖÉÁ¥†ÔºåÊâæÂà∞Á¨¨‰∏Ä‰∏™ÈúÄË¶ÅÁßªÈô§ÁöÑÂÖÉÁ¥†Á¥¢Âºï(Optimize for initial run of survivors)
        for (i = 0; i < end && !filter.test(elementAt(es, i)); i++)
            ;
        // ÈÅçÂéÜÁöÑÂêåÊó∂, ‰ªçÁÑ∂ÂÖÅËÆ∏‰ª•ÂèØÈáçÂÖ•ÊñπÂºèËÆøÈóÆÈõÜÂêàfor„ÄÅreadÔºà‰ΩÜÂÜôÂÖ•Á®ãÂ∫è‰ªçÁÑ∂Ëé∑ÂæóCMEÔºâ; 
        // Á¨¨‰∏ÄÊ¨°ÈÅçÂéÜÊü•ÊâæË¶ÅÂà†Èô§ÁöÑÂÖÉÁ¥†ÔºåÁ¨¨‰∫åÊ¨°ÈÅçÂéÜ‰ª•Áâ©ÁêÜÊñπÂºèÂà†Èô§[Êï¥‰∏™ËøáÁ®ãËÆ∞ÂΩï‰ΩøÁî®deathRow‰ΩçÂõæÁöÑÂΩ¢ÂºèÂ≠òÊîæÂåπÈÖçÊàêÂäüÁöÑÂÖÉÁ¥†ÁöÑÂÅèÁßªÈáè]
        // ÂéüÊñá: Tolerate predicates that reentrantly access the collection for, read (but writers still get CME), so traverse once to find elements to delete, a second pass to physically expunge
        if (i < end) {
            final int beg = i;
            final long[] deathRow = nBits(end - beg);
            deathRow[0] = 1L;   // ÂàùÂßãÂåñ‰ΩçÂõæ
            for (i = beg + 1; i < end; i++) // Á¨¨‰∏ÄÊ¨°ÈÅçÂéÜÊü•ÊâæË¶ÅÂà†Èô§ÁöÑÂÖÉÁ¥†
                if (filter.test(elementAt(es, i)))
                    setBit(deathRow, i - beg);
            if (modCount != expectedModCount) // Âà§Êñ≠Â§öÁ∫øÁ®ãÊòØÂê¶‰øÆÊîπ
                throw new ConcurrentModificationException();
            modCount++;
            int w = beg;
            for (i = beg; i < end; i++) // Á¨¨‰∫åÊ¨°ÈÅçÂéÜ‰ª•Áâ©ÁêÜÊñπÂºèÂà†Èô§
                if (isClear(deathRow, i - beg))
                    es[w++] = es[i];
            for (i = elementCount = w; i < end; i++)
                es[i] = null;
            return true;
        } else {
            if (modCount != expectedModCount)
                throw new ConcurrentModificationException();
            return false;
        }
    }

    // ‰ªéindex‰ΩçÁΩÆÂºÄÂßãÔºåÂ∞ÜÈõÜÂêàcÊ∑ªÂä†Âà∞Vector‰∏≠
    public synchronized boolean addAll(int index, Collection<? extends E> c) {
        if (index < 0 || index > elementCount)
            throw new ArrayIndexOutOfBoundsException(index);

        Object[] a = c.toArray();
        modCount++;
        int numNew = a.length;
        if (numNew == 0)
            return false;
        Object[] elementData = this.elementData;
        final int s = elementCount;
        if (numNew > elementData.length - s)
            elementData = grow(s + numNew);

        int numMoved = s - index;
        if (numMoved > 0)
            System.arraycopy(elementData, index,
                             elementData, index + numNew,
                             numMoved);
        System.arraycopy(a, 0, elementData, index, numNew);
        elementCount = s + numNew;
        return true;
    }

    private void shiftTailOverGap(Object[] es, int lo, int hi) {

    private class Itr implements Iterator<E> {

        int lastRet = -1; // ÊúÄÂêé‰∏ÄÊ¨°ËøîÂõûÂÖÉÁ¥†ÁöÑÁ¥¢Âºï
        int expectedModCount = modCount;

        public boolean hasNext() {
            // Ê≠§ÊñπÊ≥ïËôΩÁÑ∂Ê≤°ÊúâË¢´ÂêåÊ≠•, ‰ΩÜÊòØÂØπ‰ªñÁöÑ‰øÆÊîπÈÉΩÊòØÂú®ÂêåÊ≠•ÂÜÖÊàñËÄÖÂêåÊ≠•ÂêéÁöÑ, ÊâÄ‰ª•ËøòÊòØÂèØ‰ª•‰ΩøÁî®
            // (Ê¥ªÊ≥ºÔºå‰ΩÜÂú®ËßÑËåÉÂÜÖÔºåÂõ†‰∏∫‰øÆÊîπÊòØÂú®ÂêåÊ≠•ÂÜÖÊàñÂêåÊ≠•ÂêéÊ£ÄÊü•ÁöÑ)Racy but within spec, since modifications are checked, within or after synchronization in next/previous
            return cursor != elementCount;
        }

        public E next() {
            synchronized (Vector.this) {
                checkForComodification();
                int i = cursor;
                if (i >= elementCount)
                    throw new NoSuchElementException();
                cursor = i + 1;
                return elementData(lastRet = i);
            }
        }

        public void remove() {
            if (lastRet == -1)
                throw new IllegalStateException();
            synchronized (Vector.this) {
                checkForComodification();
                Vector.this.remove(lastRet);
                expectedModCount = modCount;
            }
            cursor = lastRet;
            lastRet = -1;
        }

        // Âè™ËÉΩÂêëÂêé‰∏ÄÊ¨°ÁöÑforEach, ‰∏ã‰∏ÄÊ¨°ÈÅçÂéÜÂàô‰∏∫Á©∫!
        @Override
        public void forEachRemaining(Consumer<? super E> action) {
            Objects.requireNonNull(action);
            synchronized (Vector.this) {
                final int size = elementCount;
                int i = cursor;
                if (i >= size) {
                    return;
                }
                final Object[] es = elementData;
                if (i >= es.length)
                    throw new ConcurrentModificationException();
                while (i < size && modCount == expectedModCount)
                    action.accept(elementAt(es, i++));
                // update once at end of iteration to reduce heap write traffic
                cursor = i;
                lastRet = i - 1;
                checkForComodification();
            }
        }

        // Âà§Êñ≠ÊòØÂê¶Â≠òÂú®Â§öÁ∫øÁ®ãÂπ∂Âèë‰øÆÊîπ Vector ÂÆû‰æã
        final void checkForComodification() {
            if (modCount != expectedModCount)
                throw new ConcurrentModificationException();
        }
    }

    final class ListItr extends Itr implements ListIterator<E> {
        ListItr(int index) {

        }

            return cursor - 1;
        }

        public E previous() {
            synchronized (Vector.this) {
                checkForComodification();
                int i = cursor - 1;
                if (i < 0)
                    throw new NoSuchElementException();
                cursor = i;
                return elementData(lastRet = i);
            }
        }

        public void set(E e) {
            if (lastRet == -1)
                throw new IllegalStateException();
            synchronized (Vector.this) {
                checkForComodification();
                Vector.this.set(lastRet, e);
            }
        }

        public void add(E e) {
            int i = cursor;
            synchronized (Vector.this) {
                checkForComodification();
                Vector.this.add(i, e);
                expectedModCount = modCount;
            }
            cursor = i + 1;
            lastRet = -1;
        }
    }

    final class VectorSpliterator implements Spliterator<E> {
        private Object[] array; // ‰∏éArrayList‰∏çÂêåÂú®‰∫éÂä†ÂÖ•‰∫Ü‰∏Ä‰∏™arrayÂºïÁî®
        private int index; // ÂΩìÂâç‰ΩçÁΩÆÔºàÂåÖÂê´ÔºâÔºåadvance/spiltÊìç‰ΩúÊó∂‰ºöË¢´‰øÆÊîπ(current index, modified on advance/split)
        private int fence; // ÁªìÊùü‰ΩçÁΩÆÔºà‰∏çÂåÖÂê´ÔºâÔºå-1Ë°®Á§∫Âà∞ÊúÄÂêé‰∏Ä‰∏™ÂÖÉÁ¥†(-1 until used; then one past last index)
        private int expectedModCount; // Áî®‰∫éÂ≠òÊîælistÁöÑmodCount(initialized when fence set)

        // ‰ªéorigin -> fenceËåÉÂõ¥ÂàõÂª∫‰∏Ä‰∏™Spliterator(Creates new spliterator covering the given range)
        VectorSpliterator(Object[] array, int origin, int fence,
                          int expectedModCount) {
            this.array = array;
            this.index = origin;
            this.fence = fence;
            this.expectedModCount = expectedModCount;
        }

        // Á¨¨‰∏ÄÊ¨°‰ΩøÁî®Êó∂ÂÆû‰æãÂåñÁªìÊùü‰ΩçÁΩÆ(initialize on first use)
        private int getFence() {
            int hi;
            // Á¨¨‰∏ÄÊ¨°ÂàùÂßãÂåñÊó∂ÔºåfenceÊâç‰ºöÂ∞è‰∫é0
            if ((hi = fence) < 0) {
                synchronized (Vector.this) {
                    array = elementData;
                    expectedModCount = modCount;
                    hi = fence = elementCount;
                }
            }
            return hi;
        }

        // ÂàÜÂâ≤list,ËøîÂõû‰∏Ä‰∏™ËøîÂõû‰∏Ä‰∏™Êñ∞ÂàÜÂâ≤Âá∫ÁöÑspilteratorÂÆû‰æã(Áõ∏ÂΩì‰∫é‰∫åÂàÜÊ≥ïÔºåËøô‰∏™ÊñπÊ≥ï‰ºöÈÄíÂΩí)
        // 1. VectorSpliteratorÊú¨Ë¥®ËøòÊòØÂØπÂéülistËøõË°åÊìç‰ΩúÔºåÂè™ÊòØÈÄöËøáindexÂíåfenceÊù•ÊéßÂà∂ÊØèÊ¨°Â§ÑÁêÜÁöÑËåÉÂõ¥
        public Spliterator<E> trySplit() {
            // hiÁªìÊùü‰ΩçÁΩÆÔºà‰∏çÂåÖÊã¨Ôºâ lo:ÂºÄÂßã‰ΩçÁΩÆ mid‰∏≠Èó¥‰ΩçÁΩÆ
            int hi = getFence(), lo = index, mid = (lo + hi) >>> 1;
            // ÂΩìlo >= midÔºå Ë°®Á§∫‰∏çËÉΩÂÜçÂàÜÂâ≤
            // ÂΩìlo < midÊó∂ÔºåË°®Á§∫ÂèØ‰ª•ÂàÜÂâ≤ÔºåÂàáÂâ≤(lo, mid)Âá∫ÂéªÔºåÂêåÊó∂Êõ¥Êñ∞index = mid
            return (lo >= mid) ? null :
                new VectorSpliterator(array, lo, index = mid, expectedModCount);
        }

        // ËøîÂõûtrueÊó∂ÔºåË°®Á§∫ÂèØËÉΩËøòÊúâÂÖÉÁ¥†Êú™Â§ÑÁêÜ, ËøîÂõûfalsaÊó∂ÔºåÊ≤°ÊúâÂâ©‰ΩôÂÖÉÁ¥†Â§ÑÁêÜ‰∫Ü
        @SuppressWarnings("unchecked")
        public boolean tryAdvance(Consumer<? super E> action) {
            Objects.requireNonNull(action);
            int i;
            if (getFence() > (i = index)) {
                index = i + 1;
                action.accept((E)array[i]);
                if (modCount != expectedModCount)
                    throw new ConcurrentModificationException();
                return true;
            }
            return false;
        }

        // È°∫Â∫èÈÅçÂéÜÂ§ÑÁêÜÊâÄÊúâÂâ©‰∏ãÁöÑÂÖÉÁ¥†
        @SuppressWarnings("unchecked")
        public void forEachRemaining(Consumer<? super E> action) {
            Objects.requireNonNull(action);
            final int hi = getFence();
            final Object[] a = array;
            int i;
            for (i = index, index = hi; i < hi; i++)
                action.accept((E) a[i]);
            if (modCount != expectedModCount)
                throw new ConcurrentModificationException();
        }

        // ‰º∞ÁÆóÂ§ßÂ∞è
        public long estimateSize() {
            return getFence() - index;
        }

        // ËøîÂõûÁâπÂæÅÂÄº
        public int characteristics() {
            return Spliterator.ORDERED | Spliterator.SIZED | Spliterator.SUBSIZED;
        }
