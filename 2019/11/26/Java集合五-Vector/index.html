    // JDK 8åæ–°åŠ å…¥, ç”¨äºä¼ å…¥ç»™å®šçš„Predicate(æ­£åˆ™è¡¨è¾¾å¼), æ¥åˆ é™¤æ»¡è¶³æ¡ä»¶çš„å…ƒç´ 
    private synchronized boolean bulkRemove(Predicate<? super E> filter) {
        int expectedModCount = modCount;
        final Object[] es = elementData;
        final int end = elementCount;
        int i;
        // ä»å¤´å¼€å§‹ç•¥è¿‡ä¸æ»¡è¶³æ–­è¨€çš„å…ƒç´ ï¼Œæ‰¾åˆ°ç¬¬ä¸€ä¸ªéœ€è¦ç§»é™¤çš„å…ƒç´ ç´¢å¼•(Optimize for initial run of survivors)
        for (i = 0; i < end && !filter.test(elementAt(es, i)); i++)
            ;
        // éå†çš„åŒæ—¶, ä»ç„¶å…è®¸ä»¥å¯é‡å…¥æ–¹å¼è®¿é—®é›†åˆforã€readï¼ˆä½†å†™å…¥ç¨‹åºä»ç„¶è·å¾—CMEï¼‰; 
        // ç¬¬ä¸€æ¬¡éå†æŸ¥æ‰¾è¦åˆ é™¤çš„å…ƒç´ ï¼Œç¬¬äºŒæ¬¡éå†ä»¥ç‰©ç†æ–¹å¼åˆ é™¤[æ•´ä¸ªè¿‡ç¨‹è®°å½•ä½¿ç”¨deathRowä½å›¾çš„å½¢å¼å­˜æ”¾åŒ¹é…æˆåŠŸçš„å…ƒç´ çš„åç§»é‡]
        // åŸæ–‡: Tolerate predicates that reentrantly access the collection for, read (but writers still get CME), so traverse once to find elements to delete, a second pass to physically expunge
        if (i < end) {
            final int beg = i;
            final long[] deathRow = nBits(end - beg);
            deathRow[0] = 1L;   // åˆå§‹åŒ–ä½å›¾
            for (i = beg + 1; i < end; i++) // ç¬¬ä¸€æ¬¡éå†æŸ¥æ‰¾è¦åˆ é™¤çš„å…ƒç´ 
                if (filter.test(elementAt(es, i)))
                    setBit(deathRow, i - beg);
            if (modCount != expectedModCount) // åˆ¤æ–­å¤šçº¿ç¨‹æ˜¯å¦ä¿®æ”¹
                throw new ConcurrentModificationException();
            modCount++;
            int w = beg;
            for (i = beg; i < end; i++) // ç¬¬äºŒæ¬¡éå†ä»¥ç‰©ç†æ–¹å¼åˆ é™¤
                if (isClear(deathRow, i - beg))
                    es[w++] = es[i];
            for (i = elementCount = w; i < end; i++)
                es[i] = null;
            return true;
        } else {
            if (modCount != expectedModCount)
                throw new ConcurrentModificationException();
            return false;
        }
    }
